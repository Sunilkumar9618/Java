//hierarichal Inheritance

class Parent{
    void parentprop(){
        System.out.println("Parent Properties");
    }
}
class Child1 extends Parent {
    void child1prop(){
        System.out.println("child Properties");
    }
}
class Child2 extends Parent {
    void child2prop(){
        System.out.println("Grand child Properties");
    }
}
public class Main
{
	public static void main(String[] args) {
		Child1 c1=new Child1();
		
		System.out.println("Child-1 Accessing Properties");
		c1.parentprop();
		c1.child1prop();
		
		System.out.println("Child-2 Accessing Properties");
		Child2 c2=new Child2();
		c2.parentprop();
		c2.child2prop();
		
	}
}
