import java.util.Scanner;
public class Main
{
	public static void main(String[] args) 
	{
	    Scanner sc=new Scanner(System.in);
	    int n=sc.nextInt();
	    int a=1;
	    for(int i=1;i<=n;i++)//rows
	    {
	       
	       for(int j=1;j<=n;j++)//collums  or   (int j=1;j<=i;j++)
	        {
	           if(i>=j){
	               System.out.printf("%02d ",a++);
	           }
	        }
	        System.out.println();
	    }

	}	
	
}
